            }
            
            const days = Math.floor(diff / 86400);
            if (days < 7) {
                return days === 1 
                    ? futureAgo + ' 1 ' + u.day
                    : futureAgo + ' ' + days + ' ' + u.days;
            }
            
            const weeks = Math.floor(days / 7);
            if (weeks < 4) {
                return weeks === 1 
                    ? futureAgo + ' 1 ' + u.week
                    : futureAgo + ' ' + weeks + ' ' + u.weeks;
            }
            
            const months = Math.floor(days / 30);
            if (months < 12) {
                return months === 1 
                    ? futureAgo + ' 1 ' + u.month
                    : futureAgo + ' ' + months + ' ' + u.months;
            }
            
            const years = Math.floor(days / 365);
            return years === 1 
                ? futureAgo + ' 1 ' + u.year
                : futureAgo + ' ' + years + ' ' + u.years;
        }
    }
